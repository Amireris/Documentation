81 def createAnnotPdf(self, myShapePdf): # inputs cfg.shapePdf, outputs cfg.annotPdf
82 '''there is a limitation on the number of vertices available to PDF annotations.
83 This can cause order geometry to get cut off.
84 If this happens, change yesBoundary = 'fixed'
85 '''
86 # input variables
87 # part 1: read geometry pdf to get the vertices and rectangle to use
88 source = PdfFileReader(open(myShapePdf,'rb'))
89 geomPage = source.getPage(0)
90 mystr = geomPage.getObject()['/Contents'].getData()
91
92 # to pinpoint the string part: 1.19997 791.75999 m 1.19997 0.19466 l 611.98627 0.19466 l 611.98627 791.75999 l 1.19997 791.75999 l
93 # the format seems to follow x1 y1 m x2 y2 l x3 y3 l x4 y4 l x5 y5 l (i.e. 302.99519 433.46045 m 308.75555 433.74844 l 309.04357 423.09304 l 303.28321 423.09304 l 302.99519 433.46045 l S)
94 previousline = ""
95 geomString = ""
96
97 # parse coordinates from pdf source code
98 try:
99 for line in mystr.split("\r\n"):
100 if bool(re.match('^\d+\..+ m .+l S$', line)) == True and bool(re.match('.+M$', previousline)) == True: # if line matches above format
101 geomString = line
102 break # get the first match only
103 previousline = line
104 if geomString == "":
105 arcpy.AddWarning("### Failed to parse geomString...")
106 raise ValueError
107 except:
108 try:
109 for line in mystr.split("\r\n"):
110 if "l S" in line and bool(re.match('.+M$', previousline)) == True:
111 geomString = line
112 break # get the first match only
113 previousline = line
114 if geomString == "":
115 arcpy.AddWarning("### Failed to parse geomString...")
116 raise
117 except:
118 geomString = mystr.split('S\r\n')[0].split('M\r\n')[1]
119 if geomString == "":
120 arcpy.AddError("### Failed to parse geomString...")
121
122 coordsString = [value for value in geomString.split(' ') if value not in ['m','l','', 'S']]
123
124 # part 2: update geometry in the map

135 updateVertices = updateVertices + ',FloatObject('+item+')'
136 updateVertices = updateVertices + "])})"
137 exec(updateVertices)

139 xcoords = []
